ALTER PROCEDURE  [dbo].[SP_GET_Employees]

@Id INT
	
	--[SP_GET_Employees] 0
As
Begin 

SET NOCOUNT ON;
	
   BEGIN TRY
       
        BEGIN TRANSACTION;

        
		
		IF(@Id = 0)
        BEGIN
		    SELECT * FROM  Employees  WITH (NOLOCK) 
			WHERE ACTIVE_FLAG = 1						
		END

		else
		BEGIN
		    SELECT * FROM  Employees  WITH (NOLOCK) 
			WHERE Id = @Id						
		END

		
	IF @@TRANCOUNT > 0
        BEGIN
            COMMIT TRANSACTION;
        END

        RETURN @Id;
    END TRY
    BEGIN CATCH
        IF @@ERROR <> 0  
        BEGIN
            ROLLBACK TRANSACTION;
            INSERT INTO tblErrorMessage (spName, errorMessage, systemDate)
            VALUES ('[SP_GET_Employees]', ERROR_MESSAGE(), GETDATE());
            SET @Id = 0;
        END
    END CATCH
END
